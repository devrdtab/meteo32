#include <Adafruit_AHTX0.h>
#include <Adafruit_BMP280.h>
#include <Arduino.h>
#include <LiquidCrystal_I2C.h>
#include <Wire.h>

void setup() {
  Serial.begin(115200);

  pinMode(17, OUTPUT);
  pinMode(18, OUTPUT);

  Serial.println("START GPIO TEST");
}

void loop() {
  Serial.println("HIGH");
  digitalWrite(17, HIGH);
  digitalWrite(18, HIGH);
  delay(1000);

  Serial.println("LOW");
  digitalWrite(17, LOW);
  digitalWrite(18, LOW);
  delay(1000);

  Serial.println(digitalRead(17));
  Serial.println(digitalRead(18));

  Serial.println("END GPIO TEST");
}
